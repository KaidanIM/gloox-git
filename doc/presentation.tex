\subsection{Extensible Markup Language}
\begin{frame}[fragile]
  \frametitle{Extensible Markup Language}
  \begin{itemize}
    \item Not a Programming Languange
    \item Rules to Describe Hierarchical Data
    \item Related to HTML
  \end{itemize}
\ \ \ \ Example:
  \begin{verbatim}
    <tag key="value">
      charachter data
    </tag>
  \end{verbatim}
\end{frame}

\subsection{Extensible Message and Presence Protocol}
\begin{frame}[fragile]
  \frametitle{Extensible Message and Presence Protocol}
  \begin{itemize}
    \item Based on XML
    \item Streaming Protocol
    \item Described in RFC 3920 and RFC 3921
  \end{itemize}
\ \ \ \ Example:
  \begin{verbatim}
    <presence
           from="jid@server.com"
           to="other@server.com">
      <status>
        online
      </status>
    </presence>
  \end{verbatim}
\end{frame}

\subsection{XMPP Client Library}
\begin{frame}
  \frametitle{XMPP Client Library}
  \begin{itemize}
    \item Encapsulates XMPP Communication
    \item Uses Observer Pattern
    \item Reusable
  \end{itemize}
\end{frame}

\subsection{Feeder Library}
\begin{frame}
  \frametitle{Feeder Library}
  \begin{itemize}
    \item Keeps Track of Connected Workers
    \item Receives Data from Host Application
    \item Distributes Data According to Availability
    \item Receives Results and Hands them to the Host Application
  \end{itemize}
\end{frame}

\subsection{Worker Library}
\begin{frame}
  \frametitle{Worker Library}
  \begin{itemize}
    \item Registers with the Feeder
    \item Receives Data, Hands it to its Host Application
    \item Receives Result from Host Application
    \item Sends the Result Back to the Feeder
  \end{itemize}
\end{frame}
